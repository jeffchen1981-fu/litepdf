- [ ] **Step 2: Render page 0 of each determinism fixture to PPM and hash**

Two non-obvious correctness points baked into the command below (plan-gate
3-lens):
- **`--render` lives in `litepdf-cli`** (`CMakeLists.txt:136`), the console
  target â€” NOT the GUI `litepdf.exe` (`src/main.cpp`, which only takes
  `--log-timings` + a path). Use `build\Release\litepdf-cli.exe`.
- **Capture binary stdout via `cmd /c`, never PowerShell `>`.** PowerShell 5.1's
  `>` is `Out-File`, which re-encodes the child's stdout as UTF-16LE (BOM +
  byte-widening + `0xFF`â†’U+FFFD), corrupting the raw P6 PPM and making the
  SHA256 meaningless â€” verified empirically in the review. `cmd /c` redirects
  raw bytes.

Run:
```powershell
$fx = @("large.pdf","cjk-zh-hant.pdf","cjk-ja.pdf","cjk-ko.pdf")
foreach ($f in $fx) {
  $out = Join-Path $env:TEMP "base-$f.ppm"
  cmd /c "build\Release\litepdf-cli.exe `"tests\fixtures\$f`" --render 0 > `"$out`""
  if ($LASTEXITCODE -ne 0) { throw "render failed for $f (exit $LASTEXITCODE)" }
  $h = (Get-FileHash -Algorithm SHA256 $out).Hash
  Write-Output "$f  $h"
}
```
Expected: four SHA256 lines, no errors. (PPM is written to `%TEMP%`, not the repo â€” keep the working tree clean.)

```cpp
// THROWAWAY spike probe â€” deleted before Stage 2. Proves the 1.27 stext
// matcher can be collected into litepdf's PageHit shape with a working
// kMaxQuads cap and a working callback abort. Linked against litepdf::mupdf
// via tests/CMakeLists.txt like the other unit tests.
#include <catch2/catch_test_macros.hpp>
#include <atomic>
#include <filesystem>
extern "C" {
#include "mupdf/fitz.h"
}

namespace {
constexpr int kMaxQuads = 256;

// Opens the existing search.pdf fixture (page 0 has "Lorem" x12) on a fresh
// context, extracts an fz_stext_page, and returns the hit count collected via
// the 1.27 matcher. abort_after >= 0 wires a callback that returns 1 (abort)
// once that many hits have been seen.
int collect(const char* needle, bool match_case, int abort_after, int* out_n) {
    // 1.27 signatures from Step 1 go here:
    //   - fz_new_context / fz_register_document_handlers
    //   - fz_open_document("tests/fixtures/search.pdf"), fz_load_page(0)
    //   - fz_new_stext_page_from_page(...)
    //   - build fz_search_options from match_case (FZ_SEARCH_EXACT vs IGNORE_CASE)
    //   - call fz_match_stext_page[_cb] filling fz_quad quads[kMaxQuads] / int marks[kMaxQuads]
    //   - if abort_after >= 0, the callback returns 1 after abort_after hits
    // Fill *out_n with the collected count, return 0 on success / non-0 on fz error.
    // (The body is written against the Step-1 signatures during the spike.)
}
} // namespace

TEST_CASE("spike: case-insensitive collects 12 Lorem on page 0", "[spike]") {
    int n = -1;
    REQUIRE(collect("lorem", /*match_case=*/false, /*abort_after=*/-1, &n) == 0);
    REQUIRE(n == 12);                 // proves collection into quads[] works
}

TEST_CASE("spike: case-sensitive does NOT match mismatched case", "[spike]") {
    int n = -1;
    REQUIRE(collect("lorem", /*match_case=*/true, /*abort_after=*/-1, &n) == 0);
    REQUIRE(n == 0);                  // proves FZ_SEARCH_EXACT path
}

TEST_CASE("spike: callback abort stops early", "[spike]") {
    int n = -1;
    REQUIRE(collect("Lorem", /*match_case=*/true, /*abort_after=*/3, &n) == 0);
    REQUIRE(n <= 3);                  // proves mid-page callback abort
}
```
